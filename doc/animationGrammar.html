BNF for Animation Grammar

TOKENS

<DEFAULT> TOKEN : {
<NUM: "0" | ["1"-"9"] (["0"-"9"])*>
| <FLOAT: ["0"-"9"] (["0"-"9"])* ("." ["0"-"9"] (["0"-"9"])*)?>
}

   
<DEFAULT> TOKEN : {
<LPAR: "{">
| <RPAR: "}">
| <STRING: (["a"-"z","A"-"Z","\'","_",".","!","?","(",")","0"-"9","\u00e0","\u00f2","\u00f9","\u00e8","\u00e9","\u00ec"])+>
}

   
<DEFAULT> SKIP : {
" "
| "\t"
| "\n"
| "\r"
| <"//" (~["\n","\r"])* ("\n" | "\r" | "\r\n")>
| <"/*" (~["*"])* "*" (~["/"] (~["*"])* "*")* "/">
}

NON-TERMINALS

Animation ::= "Animation" AnimationName AnimationDef
AnimationDef ::= "Duration:" Duration ( "SubAnimations" <LPAR> ( SubAnimation )+ <RPAR> )? "Poses" <LPAR> ( Pose )+ <RPAR> ( ( ";" | <EOF> ) )
SubAnimation ::= "Add" "Animation" AnimationName "at" "frame" Frame AnimationDef
Pose ::= "Add" "Pose" PoseName "at" "frame" Frame PoseDef
PoseDef ::= <LPAR> ( BoneName BoneDef )+ <RPAR>
BoneName ::= ( ( "HEAD:" | "CHEST:" | "R SHOULDER:" | "L SHOULDER:" | "UPPER R ARM:" | "LOWER R ARM:" | "UPPER L ARM:" | "LOWER L ARM:" | "R HIP:" | "L HIP:" | "UPPER R LEG:" | "LOWER R LEG:" | "UPPER L LEG:" | "LOWER L LEG:" ) )
BoneDef ::= Position "," Position
Frame ::= <NUM>
Position ::= "(" <FLOAT> "," <FLOAT> "," <FLOAT> ")"
AnimationName ::= ( <STRING> )+
PoseName ::= ( <STRING> )+
Duration ::= <NUM> "frames"
